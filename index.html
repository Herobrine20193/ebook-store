// Wishlist/favorites logic document.addEventListener('DOMContentLoaded', function() { function getWishlistKey() { const user = JSON.parse(localStorage.getItem('loggedInUser')); return user ? `wishlist_${user.username}` : 'wishlist_guest'; } function getWishlist() { const key = getWishlistKey(); try { return JSON.parse(localStorage.getItem(key)) || []; } catch { return []; } } function setWishlist(arr) { const key = getWishlistKey(); localStorage.setItem(key, JSON.stringify(arr)); } function updateWishlistIcons() { const wishlist = getWishlist(); document.querySelectorAll('.wishlist-btn').forEach(btn => { const bookId = btn.getAttribute('data-book-id'); const icon = btn.querySelector('i'); if (wishlist.includes(bookId)) { icon.classList.remove('fa-heart-o'); icon.classList.add('fa-heart'); btn.title = 'Remove from Wishlist'; } else { icon.classList.remove('fa-heart'); icon.classList.add('fa-heart-o'); btn.title = 'Add to Wishlist'; } }); } document.querySelectorAll('.wishlist-btn').forEach(btn => { btn.addEventListener('click', function(e) { e.preventDefault(); const bookId = btn.getAttribute('data-book-id'); let wishlist = getWishlist(); if (wishlist.includes(bookId)) { wishlist = wishlist.filter(id => id !== bookId); } else { wishlist.push(bookId); } setWishlist(wishlist); updateWishlistIcons(); }); }); updateWishlistIcons(); });
